06-事件处理

React 中的事件处理是构建交互式用户界面的核心。React 使用合成事件(SyntheticEvent)来处理用户交互，提供了跨浏览器的一致性体验。

基础事件处理

点击事件

function Button() {
  const handleClick = () => {
    console.log('按钮被点击了！');
  };

  return <button onClick={handleClick}>点击我</button>;
}

内联事件处理

function Button() {
  return (
    <button onClick={() => console.log('内联点击事件')}>
      点击我
    </button>
  );
}

常见事件类型

鼠标事件

function MouseEvents() {
  return (
    <div
      onClick={() => console.log('点击')}
      onDoubleClick={() => console.log('双击')}
      onMouseDown={() => console.log('鼠标按下')}
      onMouseUp={() => console.log('鼠标抬起')}
      onMouseEnter={() => console.log('鼠标进入')}
      onMouseLeave={() => console.log('鼠标离开')}
      onMouseMove={() => console.log('鼠标移动')}
    >
      鼠标事件测试区域
    </div>
  );
}

键盘事件

function KeyboardEvents() {
  const handleKeyDown = (e) => {
    console.log('按键按下:', e.key);
    if (e.key === 'Enter') {
      console.log('回车键被按下');
    }
  };

  return (
    <input
      type="text"
      onKeyDown={handleKeyDown}
      onKeyUp={(e) => console.log('按键抬起:', e.key)}
      onKeyPress={(e) => console.log('按键按下:', e.key)}
      placeholder="按下任意键"
    />
  );
}

表单事件

function FormEvents() {
  const [value, setValue] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('表单提交:', value);
  };

  const handleChange = (e) => {
    setValue(e.target.value);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        value={value}
        onChange={handleChange}
        onFocus={() => console.log('输入框获得焦点')}
        onBlur={() => console.log('输入框失去焦点')}
      />
      <button type="submit">提交</button>
    </form>
  );
}

    console.log('目标元素:', e.target);
    console.log('当前元素:', e.currentTarget);
    console.log('鼠标坐标:', e.clientX, e.clientY);

}

阻止默认行为

function PreventDefault() {
  const handleLinkClick = (e) => {
    e.preventDefault();
    console.log('链接点击被阻止');
  };

  const handleFormSubmit = (e) => {
    e.preventDefault();
    console.log('表单提交被阻止');
  };

  return (
    <div>
      <a href="https://example.com" onClick={handleLinkClick}>
        点击链接（不会跳转）
      </a>
      <form onSubmit={handleFormSubmit}>
        <button type="submit">提交（不会刷新页面）</button>
      </form>
    </div>
  );
}

阻止事件冒泡

function StopPropagation() {
  const handleParentClick = () => {
    console.log('父元素被点击');
  };

  const handleChildClick = (e) => {
    e.stopPropagation();
    console.log('子元素被点击（阻止冒泡）');
  };

  return (
    <div onClick={handleParentClick} style={{padding: '20px', background: '#f0f0f0'}}>
      父元素
      <button onClick={handleChildClick}>
        子元素（点击不会触发父元素事件）
      </button>
    </div>
  );
}

使用箭头函数

function ParameterPassing() {
  const handleItemClick = (id, name) => {
    console.log(`点击了项目: ${id} - ${name}`);

    { id: 1, name: '项目1' },

使用bind方法

function BindMethod() {
  const handleItemClick = (id, name, e) => {
    console.log(`点击了项目: ${id} - ${name}`, e);
  };

  return (
    <button onClick={this.handleItemClick.bind(this, 1, '项目1')}>
      点击我
    </button>
  );
}

条件事件处理

function ConditionalEvents() {
  const [isEnabled, setIsEnabled] = useState(true);

  const handleClick = () => {
    if (isEnabled) {
      console.log('按钮可用，执行点击事件');
    }
  };

  return (
    <div>
      <button
        onClick={isEnabled ? handleClick : undefined}
        disabled={!isEnabled}
      >
        {isEnabled ? '可点击' : '不可点击'}
      </button>
      <button onClick={() => setIsEnabled(!isEnabled)}>
        切换状态
      </button>
    </div>
  );
}

自定义事件

function CustomEvent() {
  const buttonRef = useRef(null);

  const dispatchCustomEvent = () => {
    const customEvent = new CustomEvent('myCustomEvent', {
      detail: { message: '自定义事件数据' }
    });
    buttonRef.current.dispatchEvent(customEvent);
  };

  const handleCustomEvent = (e) => {
    console.log('自定义事件触发:', e.detail.message);
  };

  useEffect(() => {
    const button = buttonRef.current;
    button.addEventListener('myCustomEvent', handleCustomEvent);

    return () => {
      button.removeEventListener('myCustomEvent', handleCustomEvent);
    };
  }, []);

  return (
    <div>
      <button ref={buttonRef}>监听自定义事件的按钮</button>
      <button onClick={dispatchCustomEvent}>触发自定义事件</button>
    </div>
  );
}

事件委托

function EventDelegation() {
  const handleListClick = (e) => {
    if (e.target.tagName === 'BUTTON') {
      const itemId = e.target.closest('li').dataset.id;
      console.log(`点击了按钮，项目ID: ${itemId}`);
    }
  };

  const items = Array.from({length: 100}, (_, i) => ({
    id: i + 1,
    name: `项目 ${i + 1}`
  }));

  return (
    <ul onClick={handleListClick}>
      {items.map(item => (
        <li key={item.id} data-id={item.id}>
          {item.name}
          <button>点击</button>
        </li>
      ))}
    </ul>
  );
}

性能优化

使用useCallback避免重复创建函数

function OptimizedEvents() {
  const [count, setCount] = useState(0);

  // 使用useCallback缓存事件处理函数
  const handleIncrement = useCallback(() => {
    setCount(prev => prev + 1);
  }, []);

  const handleDecrement = useCallback(() => {
    setCount(prev => prev - 1);
  }, []);

  return (
    <div>
      <p>计数: {count}</p>
      <ExpensiveButton onClick={handleIncrement} text="增加" />
      <ExpensiveButton onClick={handleDecrement} text="减少" />
    </div>
  );
}

// 使用React.memo优化子组件
const ExpensiveButton = React.memo(({ onClick, text }) => {
  console.log(`${text} 按钮重新渲染`);
  return <button onClick={onClick}>{text}</button>;
});

避免在render中创建函数

// ❌ 不推荐：每次渲染都创建新函数
function BadExample() {
  const [items, setItems] = useState([]);

  return (
    <ul>
      {items.map(item => (
        <li key={item.id}>
          <button onClick={() => handleDelete(item.id)}>
            删除
          </button>
        </li>
      ))}
    </ul>
  );
}

// ✅ 推荐：使用useCallback或提取到组件外部
function GoodExample() {
  const [items, setItems] = useState([]);

  const handleDelete = useCallback((id) => {
    setItems(prev => prev.filter(item => item.id !== id));
  }, []);

  return (
    <ul>
      {items.map(item => (
        <ListItem key={item.id} item={item} onDelete={handleDelete} />
      ))}
    </ul>
  );
}

const ListItem = React.memo(({ item, onDelete }) => (
  <li>
    <button onClick={() => onDelete(item.id)}>
      删除
    </button>
  </li>
));

移动端事件

function TouchEvents() {
  const [touchInfo, setTouchInfo] = useState('');

  const handleTouchStart = (e) => {
    const touch = e.touches[0];
    setTouchInfo(`触摸开始: (${touch.clientX}, ${touch.clientY})`);
  };

  const handleTouchMove = (e) => {
    e.preventDefault(); // 阻止页面滚动
    const touch = e.touches[0];
    setTouchInfo(`触摸移动: (${touch.clientX}, ${touch.clientY})`);
  };

  const handleTouchEnd = () => {
    setTouchInfo('触摸结束');
  };

  return (
    <div
      onTouchStart={handleTouchStart}
      onTouchMove={handleTouchMove}
      onTouchEnd={handleTouchEnd}
      style={{
        width: '200px',
        height: '200px',
        background: '#f0f0f0',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center'
      }}
    >
      <p>{touchInfo || '触摸此区域'}</p>
    </div>
  );
}

最佳实践

1. 事件处理函数命名

// ✅ 推荐：使用handle前缀
const handleClick = () => {};
const handleSubmit = () => {};
const handleChange = () => {};

// ❌ 不推荐：不清晰的命名
const click = () => {};
const doSomething = () => {};

2. 解构事件对象

// ✅ 推荐：解构需要的属性
const handleChange = ({ target: { value } }) => {
  setInputValue(value);
};

// ❌ 不推荐：使用整个事件对象
const handleChange = (e) => {
  setInputValue(e.target.value);
};

3. 使用TypeScript增强类型安全

interface ButtonProps {
  onClick: (e: React.MouseEvent<HTMLButtonElement>) => void;
  children: React.ReactNode;
}

const Button: React.FC<ButtonProps> = ({ onClick, children }) => {
  return <button onClick={onClick}>{children}</button>;
};

常见问题

问题1：事件处理函数中的this指向

class ClassComponent extends React.Component {
  constructor(props) {
    super(props);
    // 绑定this
    this.handleClick = this.handleClick.bind(this);
  }

  handleClick() {
    console.log('this:', this);
  }

  // 或者使用箭头函数
  handleClick = () => {
    console.log('this:', this);
  }

  render() {
    return <button onClick={this.handleClick}>点击</button>;
  }
}

问题2：事件处理函数中的异步操作

function AsyncEventHandler() {
  const handleClick = async (e) => {
    // 如果需要在异步操作中使用事件对象，需要持久化
    e.persist();
    
    try {
      const response = await fetch('/api/data');
      const data = await response.json();
      console.log('数据:', data);
    } catch (error) {
      console.error('错误:', error);
    }
  };

  return <button onClick={handleClick}>异步点击</button>;
}

问题3：频繁触发的事件优化

function ThrottledEvent() {
  const [scrollY, setScrollY] = useState(0);

  const throttledScrollHandler = useMemo(
    () => throttle((e) => {
      setScrollY(window.scrollY);
    }, 100),
    []
  );

  useEffect(() => {
    window.addEventListener('scroll', throttledScrollHandler);
    return () => {
      window.removeEventListener('scroll', throttledScrollHandler);
    };
  }, [throttledScrollHandler]);

  return <div>滚动位置: {scrollY}</div>;
}

// 简单的节流函数
function throttle(func, limit) {
  let inThrottle;
  return function() {
    const args = arguments;
    const context = this;
    if (!inThrottle) {
      func.apply(context, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

总结

React 的事件处理提供了：

  1. 合成事件系统 - 跨浏览器兼容性
  2. 声明式处理 - 直接在JSX中绑定事件
  3. 性能优化 - 事件委托和池化
  4. 类型安全 - TypeScript支持

掌握这些事件处理技巧，能够帮助你构建更加交互性强、性能优秀的React应用。
