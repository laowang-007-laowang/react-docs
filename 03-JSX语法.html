JSX 语法详解

什么是 JSX？

JSX (JavaScript XML) 是 React 中用于描述UI结构的语法扩展。它允许我们在JavaScript中编写类似HTML的代码。

基本语法

1. JSX 表达式

// 基本JSX
const element = <h1>Hello, World!</h1>;

// 包含JavaScript表达式
const name = 'React';
const element = <h1>Hello, {name}!</h1>;

// 表达式可以是任何有效的JavaScript
const element = <h1>当前时间: {new Date().toLocaleString()}</h1>;

2. JSX 属性

// 字符串属性
const element = <div className="container">内容</div>;

// 表达式属性
const imageUrl = 'https://example.com/image.jpg';
const element = <img src={imageUrl} alt="示例图片" />;

// 布尔属性
const element = <input type="text" disabled={true} required />;

3. JSX 子元素

// 单个子元素
const element = <div>单个文本节点</div>;

// 多个子元素
const element = (
  <div>
    <h1>标题</h1>
    <p>段落内容</p>
    <span>行内元素</span>
  </div>
);

// 子元素为表达式
const items = ['苹果', '香蕉', '橙子'];
const element = (
  <ul>
    {items.map((item, index) => (
      <li key={index}>{item}</li>
    ))}
  </ul>
);

JSX 语法规则

1. 必须有根元素

// ❌ 错误：多个根元素
const element = (
  <h1>标题</h1>
  <p>段落</p>
);

// ✅ 正确：单个根元素
const element = (
  <div>
    <h1>标题</h1>
    <p>段落</p>
  </div>
);

// ✅ 正确：使用React Fragment
const element = (
  <React.Fragment>
    <h1>标题</h1>
    <p>段落</p>
  </React.Fragment>
);

// ✅ 正确：Fragment简写
const element = (
  <>
    <h1>标题</h1>
    <p>段落</p>
  </>
);

2. 标签必须闭合

// ❌ 错误：标签未闭合
const element = <img src="image.jpg">;

// ✅ 正确：自闭合标签
const element = <img src="image.jpg" />;

// ✅ 正确：配对标签
const element = <div></div>;

3. 属性命名规则

// HTML属性转换为驼峰命名
const element = (
  <div 
    className="container"        // class → className
    htmlFor="input-id"          // for → htmlFor
    tabIndex={0}                // tabindex → tabIndex
    onClick={handleClick}       // onclick → onClick
    onMouseOver={handleHover}   // onmouseover → onMouseOver
  >
    内容
  </div>
);

在 JSX 中使用 JavaScript

1. 嵌入表达式

function UserGreeting({ user }) {
  return (
    <div>
      <h1>欢迎, {user.name}!</h1>
      <p>今天是 {new Date().toLocaleDateString()}</p>
      <p>你有 {user.messages.length} 条未读消息</p>
    </div>
  );
}

2. 条件渲染

function ConditionalRender({ isLoggedIn, user }) {
  return (
    <div>
      {/* 三元运算符 */}
      <h1>{isLoggedIn ? `欢迎, ${user.name}!` : '请先登录'}</h1>
      
      {/* 逻辑与运算符 */}
      {isLoggedIn && <button>退出登录</button>}
      
      {/* if-else 替代方案 */}
      {isLoggedIn ? (
        <div>
          <p>欢迎回来!</p>
          <button>退出</button>
        </div>
      ) : (
        <div>
          <p>请登录</p>
          <button>登录</button>
        </div>
      )}
    </div>
  );
}

3. 列表渲染

function TodoList({ todos }) {
  return (
    <ul>
      {todos.map(todo => (
        <li 
          key={todo.id}
          className={todo.completed ? 'completed' : ''}
        >
          <span>{todo.text}</span>
          <button onClick={() => deleteTodo(todo.id)}>删除</button>
        </li>
      ))}
    </ul>
  );
}

// 处理空列表
function EmptyState({ items }) {
  return (
    <div>
      {items.length > 0 ? (
        <ul>
          {items.map(item => (
            <li key={item.id}>{item.name}</li>
          ))}
        </ul>
      ) : (
        <p>暂无数据</p>
      )}
    </div>
  );
}

JSX 高级特性

1. 展开运算符

function MyComponent(props) {
  const additionalProps = {
    className: 'special',
    'data-testid': 'my-component'
  };
  
  return (
    <div {...props} {...additionalProps}>
      {/* props会被展开到div上 */}
    </div>
  );
}

// 使用组件
const element = (
  <MyComponent 
    id="test" 
    onClick={handleClick}
    style={{ color: 'red' }}
  />
);

2. 子组件传递

function Layout({ children, sidebar }) {
  return (
    <div className="layout">
      <aside>{sidebar}</aside>
      <main>{children}</main>
    </div>
  );
}

// 使用
const App = () => (
  <Layout sidebar={<Sidebar />}>
    <h1>主要内容</h1>
    <p>这里是页面内容</p>
  </Layout>
);

3. Render Props 模式

function DataProvider({ render }) {
  const [data, setData] = useState(null);
  
  useEffect(() => {
    fetchData().then(setData);
  }, []);
  
  return render(data);
}

// 使用
const App = () => (
  <DataProvider 
    render={data => (
      <div>
        {data ? (
          <div>数据: {JSON.stringify(data)}</div>
        ) : (
          <div>加载中...</div>
        )}
      </div>
    )}
  />
);

样式处理

1. 内联样式

function StyledComponent() {
  const styles = {
    container: {
      backgroundColor: '#f0f0f0',
      padding: '20px',
      borderRadius: '8px'
    },
    title: {
      color: '#333',
      fontSize: '24px',
      marginBottom: '16px'
    }
  };
  
  return (
    <div style={styles.container}>
      <h1 style={styles.title}>标题</h1>
      <p style={{ color: 'blue', fontWeight: 'bold' }}>内容</p>
    </div>
  );
}

2. CSS 类名

// 静态类名
function Component() {
  return <div className="container active">内容</div>;
}

// 动态类名
function DynamicClass({ isActive, isDisabled }) {
  const classNames = [
    'base-class',
    isActive && 'active',
    isDisabled && 'disabled'
  ].filter(Boolean).join(' ');
  
  return <div className={classNames}>内容</div>;
}

// 使用第三方库 (classnames)
import classNames from 'classnames';

function Component({ isActive, isDisabled }) {
  return (
    <div className={classNames('base-class', {
      active: isActive,
      disabled: isDisabled
    })}>
      内容
    </div>
  );
}

事件处理

1. 基本事件处理

function EventExample() {
  const handleClick = (e) => {
    e.preventDefault();
    console.log('按钮被点击了');
  };
  
  const handleChange = (e) => {
    console.log('输入值:', e.target.value);
  };
  
  return (
    <div>
      <button onClick={handleClick}>点击我</button>
      <input onChange={handleChange} placeholder="输入文本" />
    </div>
  );
}

2. 传递参数

function ListComponent({ items, onItemClick }) {
  return (
    <ul>
      {items.map(item => (
        <li key={item.id}>
          <button onClick={() => onItemClick(item.id)}>
            {item.name}
          </button>
          {/* 或者使用bind */}
          <button onClick={onItemClick.bind(null, item.id)}>
            {item.name}
          </button>
        </li>
      ))}
    </ul>
  );
}

常见错误和解决方案

1. key 属性

// ❌ 错误：缺少key或使用index作为key
{items.map((item, index) => (
  <div>{item.name}</div>  // 缺少key
))}

{items.map((item, index) => (
  <div key={index}>{item.name}</div>  // 使用index作为key可能有问题
))}

// ✅ 正确：使用唯一标识符作为key
{items.map(item => (
  <div key={item.id}>{item.name}</div>
))}

2. 事件处理器

// ❌ 错误：直接调用函数
<button onClick={handleClick()}>点击</button>

// ✅ 正确：传递函数引用
<button onClick={handleClick}>点击</button>

// ✅ 正确：传递匿名函数
<button onClick={() => handleClick(param)}>点击</button>

3. 条件渲染陷阱

// ❌ 可能出错：falsy值会被渲染
{items.length && <div>有 {items.length} 个项目</div>}

// ✅ 安全做法：显式boolean转换
{items.length > 0 && <div>有 {items.length} 个项目</div>}
{Boolean(items.length) && <div>有 {items.length} 个项目</div>}

最佳实践

  1. 组件名大写: 组件名必须以大写字母开头
  2. 使用Fragment: 避免不必要的包装div
  3. Key属性: 列表渲染时使用唯一且稳定的key
  4. 事件处理: 避免在JSX中直接调用函数
  5. 条件渲染: 注意falsy值的渲染行为
  6. 样式组织: 合理组织CSS类名和内联样式
