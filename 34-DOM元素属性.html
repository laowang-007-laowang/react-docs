34-DOM元素属性

React支持大部分HTML标准属性，但有一些特殊的命名约定和行为。本文档详细介绍React中DOM元素的属性使用方法。

属性命名约定

// React中的属性命名（camelCase）
function AttributeNaming() {
  return (
    <div
      className="box"           // class -> className
      htmlFor="input-id"        // for -> htmlFor
      tabIndex={0}              // tabindex -> tabIndex
      autoComplete="off"        // autocomplete -> autoComplete
      autoFocus={true}          // autofocus -> autoFocus
      contentEditable={true}    // contenteditable -> contentEditable
      spellCheck={false}        // spellcheck -> spellCheck
      readOnly={true}           // readonly -> readOnly
      maxLength={100}           // maxlength -> maxLength
      minLength={5}             // minlength -> minLength
      rowSpan={2}               // rowspan -> rowSpan
      colSpan={3}               // colspan -> colSpan
      useMap="#map"             // usemap -> useMap
      accessKey="k"             // accesskey -> accessKey
      contentEditable="true"    // contenteditable -> contentEditable
    />
  )
}

// 访问数据属性
function AccessDataAttributes() {
  const handleClick = (event) => {
    const element = event.currentTarget
    
    // 使用 dataset 访问数据属性
    console.log(element.dataset.userId)        // "123"
    console.log(element.dataset.role)          // "admin"
    console.log(element.dataset.createdAt)     // "2023-01-01"
    console.log(element.dataset.config)        // '{"theme": "dark", "lang": "zh"}'
    console.log(element.dataset.analyticsEvent) // "user-click"
    
    // 解析JSON数据
    const config = JSON.parse(element.dataset.config)
    console.log(config.theme) // "dark"
  }
  
  return (
    <div
      data-user-id="123"
      data-role="admin"
      data-created-at="2023-01-01"
      data-config='{"theme": "dark", "lang": "zh"}'
      data-analytics-event="user-click"
      onClick={handleClick}
    >
      点击获取数据属性
    </div>
  )
}

危险属性

function DangerousAttributes() {
  const htmlContent = '<p>这是<strong>粗体</strong>文本</p>'
  const sanitizedContent = sanitizeHtml(htmlContent) // 需要使用消毒库
  
  return (
    <div>
      {/* 危险：直接设置HTML内容 */}
      <div dangerouslySetInnerHTML={{ __html: htmlContent }} />
      
      {/* 安全：使用消毒后的HTML内容 */}
      <div dangerouslySetInnerHTML={{ __html: sanitizedContent }} />
      
      {/* 推荐：使用组件渲染 */}
      <div>
        <p>这是<strong>粗体</strong>文本</p>
      </div>
    </div>
  )
}

属性传递

function AttributeSpread() {
  const baseProps = {
    className: 'base-class',
    'data-testid': 'test-element',
    role: 'button',
    tabIndex: 0
  }
  
  const additionalProps = {
    'aria-label': '操作按钮',
    onClick: handleClick
  }
  
  return (
    <div>
      {/* 展开属性 */}
      <button {...baseProps} {...additionalProps}>
        按钮
      </button>
      
      {/* 属性合并 */}
      <div {...baseProps} className={`${baseProps.className} extra-class`}>
        合并类名
      </div>
    </div>
  )
}

// 自定义组件属性传递
function CustomButton({ children, className, ...restProps }) {
  return (
    <button
      className={`custom-button ${className || ''}`}
      {...restProps}
    >
      {children}
    </button>
  )
}

总结

React DOM属性的关键点：

  1. 命名约定 - 使用camelCase命名
  2. 布尔属性 - 可以省略值或使用布尔值

  4. 事件属性 - 使用React合成事件系统
  5. 数据属性 - 使用data-*格式
  6. ARIA属性 - 提供无障碍支持
  7. 引用属性 - 使用ref获取DOM元素
