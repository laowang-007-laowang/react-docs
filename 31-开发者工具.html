function useCounter() {
  const [count, setCount] = useState(0)
  const [history, setHistory] = useState([0])
  
  const increment = () => {
    setCount(prev => {
      const newCount = prev + 1
      setHistory(h => [...h, newCount])
      return newCount
    })
  }
  
  // 在DevTools中可以看到所有Hooks的状态
  return { count, history, increment }
}

错误边界调试

class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props)
    this.state = { hasError: false, error: null }
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, error }
  }

  componentDidCatch(error, errorInfo) {
    // 详细的错误信息
    console.error('Error Boundary caught an error:', error, errorInfo)
    
    // 可以发送错误报告
    this.logErrorToService(error, errorInfo)
  }

  logErrorToService(error, errorInfo) {
    // 发送到错误监控服务
    if (process.env.NODE_ENV === 'production') {
      // Sentry, LogRocket等
    }
  }

  render() {
    if (this.state.hasError) {
      return (
        <div style={{ padding: '20px', border: '1px solid red' }}>
          <h2>出现错误</h2>
          <details style={{ whiteSpace: 'pre-wrap' }}>
            {this.state.error && this.state.error.toString()}
          </details>
        </div>
      )
    }

    return this.props.children
  }
}

function useDebugValue(value, label) {

function useCounter() {
  const [count, setCount] = useState(0)
  
  // 在React DevTools中显示调试信息

}

// 性能调试Hook
function useRenderCount() {
  const renderCount = useRef(0)
  
  useEffect(() => {
    renderCount.current++
  })
  
  useDebugValue(renderCount.current, 'Render Count')
  
  return renderCount.current
}

自定义性能监控

// 性能监控Hook
function usePerformanceMonitor(componentName) {
  const renderStart = useRef()
  const renderCount = useRef(0)
  
  useEffect(() => {
    renderStart.current = performance.now()
    renderCount.current++
  })
  
  useEffect(() => {
    const renderEnd = performance.now()
    const renderTime = renderEnd - renderStart.current
    
    if (renderTime > 16) { // 超过16ms的渲染
      console.warn(`${componentName} 渲染时间过长: ${renderTime}ms`)
    }
    
    if (renderCount.current > 10) { // 频繁渲染
      console.warn(`${componentName} 渲染次数过多: ${renderCount.current}`)
    }
  })
}

// 使用示例
function MyComponent() {
  usePerformanceMonitor('MyComponent')
  
  return <div>内容</div>
}

自动化脚本

// scripts/component-generator.js
const fs = require('fs')
const path = require('path')

const componentTemplate = (name) => `
import React from 'react'
import './${name}.css'

function ${name}() {
  return (
    <div className="${name.toLowerCase()}">
      <h1>${name}</h1>
    </div>
  )
}

export default ${name}
`

const cssTemplate = (name) => `
.${name.toLowerCase()} {
  /* Add your styles here */
}
`

const testTemplate = (name) => `
import React from 'react'
import { render, screen } from '@testing-library/react'
import ${name} from './${name}'

describe('${name}', () => {
  test('renders correctly', () => {
    render(<${name} />)
    expect(screen.getByText('${name}')).toBeInTheDocument()
  })
})
`

function generateComponent(name) {
  const dir = path.join(__dirname, `../src/components/${name}`)
  
  if (!fs.existsSync(dir)) {
    fs.mkdirSync(dir, { recursive: true })
  }
  
  fs.writeFileSync(path.join(dir, `${name}.jsx`), componentTemplate(name))
  fs.writeFileSync(path.join(dir, `${name}.css`), cssTemplate(name))
  fs.writeFileSync(path.join(dir, `${name}.test.jsx`), testTemplate(name))
  
  console.log(`Component ${name} generated successfully!`)
}

const componentName = process.argv[2]
if (!componentName) {
  console.error('Please provide a component name')
  process.exit(1)
}

generateComponent(componentName)

// 性能调试工具
const perf = {
  start: (label) => {
    if (process.env.NODE_ENV === 'development') {
      console.time(label)
    }
  },
  
  end: (label) => {
    if (process.env.NODE_ENV === 'development') {
      console.timeEnd(label)
    }
  },
  
  measure: (fn, label) => {
    return (...args) => {
      perf.start(label)
      const result = fn(...args)
      perf.end(label)
      return result
    }
  }
}
